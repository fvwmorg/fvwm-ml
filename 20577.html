# FvwmButtons ConfigFile

Fore Black
Back gray70
Font -adobe-helvetica-bold-r-*-*-10-*-*-*-*-*-*-*
Geometry 805x57+0-0
Rows 1
Padding 0 0
Frame 1

(Container)
  (1x2, Frame 1) - klo_inet.xpm Function fvwm2b_nrm
  (2x1, Title Menue, Action(mouse1) 'PopUp Main-Menu', Action(mouse2) Nop, Action(mouse3) 'WindowList')
    - mini-display.xpm Refresh
    - /root/_pixmaps/bar/max.xpm Function "maximize_func"
(End)

(Container(Rows 3))
 (2x1, Fore rgb:d580/0000/b9bf) xterm - Exec exec /usr/X11R6/bin/xterm -T xterm &
 (1x1, Fore rgb:8b85/0000/7961) ftp - Exec exec /usr/X11R6/bin/xterm -T 'xterm: ftp' -e /usr/bin/ftp &
 (1x1, Fore rgb:8b85/0000/7961) telnet - Exec exec /usr/X11R6/bin/xterm -T 'xterm: telnet' -e /usr/bin/telnet &
 (2x1, Fore rgb:d580/0000/b9bf) eIrc - Exec exec eIrc &
(End)

- mc.xpm Exec /usr/X11R6/bin/xterm -T mc -n mc -e mc -c &
- netscape_3d.xpm Popup FvwmNetscape

(5x1, Container(Rows 3))
 (3x1, Frame 1, Back grey78, Fore darkblue,Title (Left)" PPPCOSTS :") -
 (1x1, Back rgb:10/90/10, Fore white, Font fixed) ppp-GO - Exec /root/bin/scripts/ppp-go &
 (1x1, Back rgb:90/10/10, Fore white, Font fixed) ppp-OFF - Exec /root/bin/scripts/ppp-off &
 (5x2, Frame 1, Back grey78) - - Swallow "pppcosts" Exec rxvt -fn fixed -bg grey78 -cr grey70 -fg darkblue +sb -geometry 56x2+100+650 -e pppcosts &
(End)
 
(2x1) - whatever Swallow "FvwmPager" Module FvwmPager 0 0 &

- whatever Swallow "asclock" Exec /usr/X11R6/bin/asclock -shape -led yellow -noblink -position +100+650 &
