check this out :)

---------------------
.fvwm2rc:
---------------------

###################### fvwm button-bar
##########################################
# FvwmMenues
AddToMenu FvwmNetscape
+ "*klo_navigator.xpm*Navigator" Exec netscape -geometry 750x500 &
+ "*klo_mail.xpm*Messanger" Exec netscape -mail -geometry 750x500 &
+ "*klo_center.xpm* Center " Exec netscape -news -geometry 350x500
&
+ "" Nop
                                    
*FvwmButtonsFile /root/.fvwm2buttonsrc

-----------------------------
.fvwm2buttonsrc:
-----------------------------

# FvwmButtons ConfigFile

Fore Black
Back gray70
Font -adobe-helvetica-bold-r-*-*-10-*-*-*-*-*-*-*
Geometry +15-0
Rows 1

( Container)
  (2x1, Icon mini-exp.xpm, Title(Side) Menue, Action 'Menu Main-Menu')
    - mini-display.xpm Refresh
    - mini-exclam.xpm Restart fvwm2
(End)
- cli.xpm Exec "rxvt" rxvt -bg black -fg white -T rxvt &
- Shell2.linux.xpm Exec "xterm" xterm -sb -sl 200 &
- Text.xpm Exec "rxvt" rxvt -bg black -fg white -fn fixed -T
Messages-Log -e tail -f /var/log/messages
- mc.xpm Exec "mc" xterm -T mc -n mc -e mc -c &
- FileManager.xpm Exec "X-Files" X-Files &
- xv_3d.xpm Exec "xv" xv &
- rcalc_3d.xpm Exec "xcalc" xcalc &
- netscape_3d.xpm Popup FvwmNetscape
(2x1) - whatever Swallow "FvwmPager" Module FvwmPager 0 0 &
- whatever Swallow "asclock" Exec exec asclock -shape -led green
-position +1000 +100 &
