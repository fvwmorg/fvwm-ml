+extern int sys_nerr;
+extern char *sys_errlist[];
+
+/* strerror
+ *
+ * Return string corresponding to errno
+ * For systems which lack this function.
+ */
+const char *
+strerror(int errnum)
+{
+ static char unknown[] = "Unknown error";
+
+ return ((unsigned int)errnum < sys_nerr) ? sys_errlist[errnum] : unknown;
+}
