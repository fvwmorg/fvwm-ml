try this (I've got this in my .bashrc):

#set the title of an xterm to show $PWD

if [ "$TERM" = "xterm" ]; then
  function cd ()
  {
    builtin cd $*
    xtitle "`whoami`_at_${HOSTNAME%%.*}:$PWD (`date`)"
  }
  else
  function cd ()
  {
    builtin cd $*
  }
fi

xtitle()
{
    echo -e -n "\033]2;$*\007"
}
