# FvwmButtons ConfigFile

Fore Black
Back gray70
Font -adobe-helvetica-bold-r-*-*-10-*-*-*-*-*-*-*
Geometry 805x57+0-0
Rows 1
Padding 0 0
Frame 1


(Container)
  (1x2) - klo_inet.xpm Function fvwm2b_nrm
  (2x1, Title Menue, Action(mouse1) 'PopUp Main-Menu', Action(mouse2) Nop, Action(mouse3) 'WindowList')
    - mini-display.xpm Refresh
    - /root/_pixmaps/bar/max.xpm Function "maximize_func"
(End)

#- Shell2.linux.xpm Exec xterm -sb -sl 200 -bg gray80 -fg black &

(1x1, Container(Rows 3))
 xterm - Exec xterm -T xterm &
 ftp - Exec exec xterm -T ftp -e ftp &
 telnet - Exec xterm -T telnet -e telnet &
(End)


- mc.xpm Exec xterm -T mc -n mc -e mc -c &
- netscape_3d.xpm Popup FvwmNetscape

(5x1, Container(Rows 3))
 (3x1, Frame 1, Back grey78, Fore darkblue,Title (Left)" PPPCOSTS :") -
 (1x1) ppp-GO - Exec /root/bin/scripts/ppp-go &
 (1x1) ppp-OFF - Exec /root/bin/scripts/ppp-off &
 (5x2, Frame 1, Back grey78) - - Swallow "pppcosts" Exec rxvt -fn fixed -bg grey78 -cr grey78 -fg darkblue +sb -geometry 56x2+100+650 -e pppcosts &
(End)
 
(2x1) - whatever Swallow "FvwmPager" Module FvwmPager 0 0 &

- whatever Swallow "asclock" Exec exec asclock -shape -led yellow -noblink -position +100+650 &
