###################### button-bar ##########################################
*FvwmButtonsFore Black
*FvwmButtonsBack gray70
#908090
*FvwmButtonsFont -adobe-helvetica-bold-r-*-*-10-*-*-*-*-*-*-*
*FvwmButtonsGeometry +15-0
*FvwmButtonsRows 1

*FvwmButtons(1x1 Container(Rows 2)
  *FvwmButtons - mini-bomb.xpm Destroy
  *FvwmButtons - /usr/X11R6/include/X11/pixmaps/bar/max.xpm Function maximize_func
  *FvwmButtons - mini-display.xpm Refresh
  *FvwmButtons - mini-exclam.xpm Restart fvwm2
*FvwmButtons(End)

*FvwmButtons - cli.xpm Exec "rxvt" rxvt -bg black -fg white -T rxvt &
*FvwmButtons - xterm.next.xpm Exec "xterm" xterm -sb -sl 200 &
*FvwmButtons - Text.xpm Exec "xterm" xterm -bg azure2 -fg black -T Messages-Log -e tail -f
/var/adm/messages &

*FvwmButtons - mc.xpm Exec "mc" rxvt -bg black -fg white -T mc -n mc -e mc -c &
*FvwmButtons - FileManager.xpm Exec "X-Files" X-Files &

*FvwmButtons - xv_3d.xpm Exec "xv" xv &
*FvwmButtons - rcalc_3d.xpm Exec "xcalc" xcalc &
*FvwmButtons - netscape_3d.xpm Exec "Netscape" /root/Programme/netscape/netscape -geometry 750x500 &
*FvwmButtons(2x1) - whatever Swallow "FvwmPager" Module FvwmPager 0 0 &
*FvwmButtons(1x1) - whatever Swallow "asclock" Exec asclock -noblink -shape -led green -position
+100+600 &
